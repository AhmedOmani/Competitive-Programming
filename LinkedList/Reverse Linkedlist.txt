struct node{

    int value ;
    node* next ;
};

node* head = NULL ;

void InsertNode(int val) {

    node* NewNode , *last ;
    NewNode = new node ;

    NewNode->value = val ;

    if (head == NULL) {

        head = NewNode ;
        NewNode->next = NULL;

    } else {

        last = head ;

        while (last->next != NULL)
            last = last->next ;

        last->next = NewNode ;
        NewNode->next = NULL;

    }

}
