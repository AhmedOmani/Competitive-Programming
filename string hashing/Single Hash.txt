int subtraction(int a , int b) {

int add(int a , int b){
    return ((a % mod) + (b % mod)) % mod ;
}

int mult(int a , int b) {
    return ((a % mod) * (b % mod)) % mod ;
}

int FastPower(int a , int b) {

    if(!b)
        return 1 ;

    int temp = FastPower(a , b / 2) ;

    temp = mult(temp , temp) ;

    if (b & 1)
        temp = mult(temp , a) ;

    return temp ;
}


void BuildPower() {

    Power[0] = 1 , inv[0] = 1;

    int mult_inv = FastPower(p , mod - 2) ;

    return Prefix[s.size()] ;
